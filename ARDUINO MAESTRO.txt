//Arduino Maestro/Transmitter

void setup() {
  Serial.begin(9600);
}

void loop() {
  int lectura1 = analogRead(A0);
  int lectura2 = analogRead(A1);
  int lectura3 = analogRead(A2);
  int lectura4 = analogRead(A3);
  int lectura5 = analogRead(A4);
  int lectura6 = analogRead(A5);

  Serial.write(angulo1);
  Serial.write(angulo2);
  Serial.write(angulo3);
  Serial.write(angulo4);
  Serial.write(angulo5);
  Serial.write(angulo6);

  delay(50);
}
